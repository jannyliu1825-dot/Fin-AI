    <script>
        // 隱藏載入動畫
        function hideLoader() {
            document.getElementById('loader').style.display = 'none';
        }

        // 切換頁籤邏輯
        function switchTab(tabId) {
            const btn1 = document.getElementById('btn-tab1');
            const btn2 = document.getElementById('btn-tab2');
            const content1 = document.getElementById('content-tab1');
            const content2 = document.getElementById('content-tab2');

            // 基礎樣式類別 (包含圖標和其他共用設定)
            const activeClass = 'tab-active border px-6 py-3 rounded-xl font-medium transition-all duration-300 flex items-center gap-2.5 shadow-lg transform active:scale-95';
            const inactiveClass = 'tab-inactive border px-6 py-3 rounded-xl font-medium transition-all duration-300 flex items-center gap-2.5 shadow-lg transform active:scale-95';

            if (tabId === 'tab1') {
                // 更新按鈕狀態
                btn1.className = activeClass;
                btn2.className = inactiveClass;

                // 顯示 Tab 1, 隱藏 Tab 2
                content1.classList.replace('z-0', 'z-10');
                content1.classList.replace('opacity-0', 'opacity-100');
                content1.classList.remove('pointer-events-none');

                content2.classList.replace('z-10', 'z-0');
                content2.classList.replace('opacity-100', 'opacity-0');
                content2.classList.add('pointer-events-none');
            } else {
                // 更新按鈕狀態
                btn2.className = activeClass;
                btn1.className = inactiveClass;

                // 顯示 Tab 2, 隱藏 Tab 1
                content2.classList.replace('z-0', 'z-10');
                content2.classList.replace('opacity-0', 'opacity-100');
                content2.classList.remove('pointer-events-none');

                content1.classList.replace('z-10', 'z-0');
                content1.classList.replace('opacity-100', 'opacity-0');
                content1.classList.add('pointer-events-none');
            }
        }
    </script>
